2) Upgraded end‑to‑end pipeline (v2.0) — Replit‑ready

One run = 4 stages: Ingest ➜ Score ➜ Enrich ➜ Export (with an Incremental Sync loop so you can run it daily/weekly without re‑processing everything).

Stage A — Ingest (Socrata API; incremental & streaming)

Pull Debtor rows from 8upq-58vz and Secured‑Party rows from ap62-sav4 using SODA API with server‑side filters:

recordstatus='Active'

upper(state) in ('CO','COLORADO') or address state fields as present

Optional: date filter when the dataset exposes a filing/action date (e.g., last N months).

Use $select, $where, $limit/$offset or $order for paging, and If‑Modified‑Since style logic to do incremental pulls. 
Socrata Developers

Keep the dataset IDs configurable; Socrata docs + SDK examples show both raw HTTP and Python client usage. 
Socrata Support

(Optional) Pull Collateral data (4am6-w6u4) to detect “accounts receivable” or “inventory” strings as a bonus MCA signal. 
Colorado Information Marketplace

Stage B — Scoring (business‑level, de‑duped)

Normalize & join the two feeds on filing ID.

Construct features per business (normalize by name + ZIP with fuzzy back‑stop):

has_bank: secured‑party contains tokens like " bank", "credit union", " national association", " n.a", " fsb".

has_equipment: contains brand tokens (e.g., deere, caterpillar, dll, komatsu, kubota, volvo financial, vermeer, doosan).

has_irs, has_sba: negatives.

Counts: active_filing_count, secured_party_count.

Recency: last_filing_date (if a file date is present in your source; when missing, omit or proxy with most recent action).

Exclusions: government/public entities (name contains city of, county of, authority, school district, university, state of, etc.) and entities with SBA/IRS flags.

Score (weights tunable by YAML/env):

score =
  + 25 * has_bank
  + 25 * has_equipment
  + 20 * min(secured_party_count,10)/10
  + 20 * min(active_filing_count,10)/10
  + 10 * recency_bucket   # if available (e.g., 0..10)
  - 60 * has_irs
  - 40 * has_sba


De‑dupe to one row per business: canonicalize (name, zip); then fuzzy merge near‑duplicates (e.g., “Inc.” vs “Incorporated”) using RapidFuzz/Jaro‑Winkler; keep the highest‑scoring record.

Socrata lets you scale to large pulls and do server‑side filtering; use $where to keep memory low, and page in batches. 
Socrata Developers

Query Places Text Search with "business name + city + Colorado"; take the top match in CO, then Place Details → phone, website, formatted address, place_id. This is far more reliable than scraping random directories at scale.

main.py (paste into Replit)

import os, sys, time, json, math, sqlite3, yaml, re
from datetime import datetime, timedelta
from urllib.parse import urlencode
import requests
import pandas as pd
from rapidfuzz import fuzz, process

# ---------- load config/state ----------
cfg = yaml.safe_load(open("config.yaml", "r"))
STATE_DB = cfg["output"]["state_db"]
CSV_OUT = cfg["output"]["csv"]

def db():
    conn = sqlite3.connect(STATE_DB)
    conn.execute("""CREATE TABLE IF NOT EXISTS state (k TEXT PRIMARY KEY, v TEXT)""")
    conn.execute("""CREATE TABLE IF NOT EXISTS seen (k TEXT PRIMARY KEY, seen_at TEXT)""")
    return conn

def get_state(k, default=""):
    with db() as c:
        cur = c.execute("SELECT v FROM state WHERE k=?", (k,))
        row = cur.fetchone()
        return row[0] if row else default

def set_state(k, v):
    with db() as c:
        c.execute("REPLACE INTO state(k,v) VALUES(?,?)", (k, v))

def mark_seen(key):
    with db() as c:
        c.execute("REPLACE INTO seen(k,seen_at) VALUES(?,?)", (key, datetime.utcnow().isoformat()))

def is_seen(key):
    with db() as c:
        cur = c.execute("SELECT 1 FROM seen WHERE k=?", (key,))
        return cur.fetchone() is not None

# ---------- Socrata helpers ----------
SOC_DOMAIN = cfg["socrata"]["domain"]
DEBTOR_DS = cfg["socrata"]["debtor_dataset"]
SECURED_DS = cfg["socrata"]["secured_dataset"]
APP_TOKEN = cfg["socrata"].get("app_token") or ""
PAGE = int(cfg["socrata"]["page_size"])

HEADERS = {"X-App-Token": APP_TOKEN} if APP_TOKEN else {}

def soda_get(ds, where=None, select=None, order=None, limit=5000, offset=0):
    params = {"$limit": limit, "$offset": offset}
    if where: params["$where"] = where
    if select: params["$select"] = select
    if order: params["$order"] = order
    url = f"https://{SOC_DOMAIN}/resource/{ds}.json?{urlencode(params)}"
    r = requests.get(url, headers=HEADERS, timeout=60)
    r.raise_for_status()
    return r.json()

def iter_socrata(ds, where=None, select=None, order=None):
    off = 0
    while True:
        chunk = soda_get(ds, where=where, select=select, order=order, limit=PAGE, offset=off)
        if not chunk: break
        yield chunk
        off += PAGE

def norm_cols(df):
    df.columns = [c.strip().lower() for c in df.columns]
    return df

def to_date(s):
    try:
        return pd.to_datetime(s, errors="coerce").date()
    except Exception:
        return None

# ---------- pull debtor & secured ----------
def build_filters():
    parts = ["lower(recordstatus) = 'active'"]
    # Try common state fields; Socrata schema differs across exports — we'll OR the possibilities
    state_cols = ["state", "debtor_state", "address_state"]
    state_where = "(" + " OR ".join([f"upper({c}) in ('CO','COLORADO')" for c in state_cols]) + ")"
    parts.append(state_where)
    # lookback if a date exists (we can't assume column name — handled later at feature time)
    return " AND ".join(parts)

def pull_debtors():
    where = build_filters()
    # pull light set of columns broadly named; Socrata will ignore missing fields
    select = ",".join([
        "fileid", "filenumber", "fileno", "recordnumber",
        "organizationname", "orgname",
        "lastname","firstname","middlename","suffix",
        "address1","address2","city","state","zipcode","zip","postalcode",
        "recordstatus","filedate","filingdate","recorddate","eventdate"
    ])
    rows = []
    for chunk in iter_socrata(DEBTOR_DS, where=where, select=select):
        rows.extend(chunk)
    return pd.DataFrame(rows)

def pull_secured():
    where = "lower(recordstatus) = 'active'"
    select = ",".join([
        "fileid", "filenumber","fileno","recordnumber",
        "organizationname","orgname","lastname","firstname","recordstatus"
    ])
    rows = []
    for chunk in iter_socrata(SECURED_DS, where=where, select=select):
        rows.extend(chunk)
    return pd.DataFrame(rows)

# ---------- feature engineering ----------
TERMS = cfg["terms"]
EX_EXCLUDE = [t.lower() for t in cfg["filters"]["exclude_terms"]]

def any_term(s, toks):
    s = (s or "").lower()
    return int(any(t in s for t in toks))

def secparty_flags(sp_blob):
    sp = (sp_blob or "").lower()
    return {
        "has_bank": any_term(sp, TERMS["bank"]),
        "has_equipment": any_term(sp, TERMS["equipment"]),
        "has_irs": any_term(sp, TERMS["irs"]),
        "has_sba": any_term(sp, TERMS["sba"]),
        "count": 0 if not sp else len([x for x in sp.split(";") if x.strip()]),
    }

def best_fileid(df):
    for c in ["fileid","filenumber","fileno","recordnumber"]:
        if c in df.columns: return c
    return None

def best_orgname(row):
    for c in ["organizationname","orgname"]:
        if c in row and row[c]: return row[c]
    # fallback to personal name if org missing
    bits = [row.get("lastname",""), row.get("firstname",""), row.get("middlename",""), row.get("suffix","")]
    nm = " ".join([b for b in bits if b]).strip()
    return nm

def build_business_rollup(deb_df, sec_df):
    deb_df = norm_cols(deb_df); sec_df = norm_cols(sec_df)
    fid_deb = best_fileid(deb_df); fid_sec = best_fileid(sec_df)
    if not fid_deb or not fid_sec: raise SystemExit("No filing id columns")

    # aggregate secured party names per filing id
    if "organizationname" in sec_df.columns:
        sec_df["sp_name"] = sec_df["organizationname"].fillna("").str.strip()
    elif "orgname" in sec_df.columns:
        sec_df["sp_name"] = sec_df["orgname"].fillna("").str.strip()
    else:
        lf = sec_df.get("lastname","").fillna("")
        ff = sec_df.get("firstname","").fillna("")
        sec_df["sp_name"] = (lf + " " + ff).str.strip()

    sp = sec_df.groupby(fid_sec)["sp_name"].apply(
        lambda s: "; ".join(sorted(set([x for x in s if isinstance(x,str) and x])))
    ).reset_index(name="secured_parties")

    j = deb_df.merge(sp, how="left", left_on=fid_deb, right_on=fid_sec)
    j["debtor_name"] = j.apply(best_orgname, axis=1)
    # address
    j["city"] = j.get("city","")
    j["state"] = j.get("state","")
    j["zip"] = j.get("zipcode", j.get("zip", j.get("postalcode","")))
    j["file_date"] = pd.to_datetime(j.get("filedate", j.get("filingdate", j.get("recorddate", j.get("eventdate","")))), errors="coerce")

    # remove obvious gov/public entities
    bad = j["debtor_name"].fillna("").str.lower().apply(lambda x: any(t in x for t in EX_EXCLUDE))
    j = j[~bad]

    # collapse to (name,zip)
    g = j.groupby(["debtor_name","zip"], dropna=False).agg(
        city=("city","first"),
        state=("state","first"),
        secured_parties=("secured_parties", lambda s: "; ".join(sorted(set("; ".join([x for x in s if isinstance(x,str) and x]).split("; ")))[:12])),
        active_filing_count=("debtor_name","count"),
        last_filing_date=("file_date","max")
    ).reset_index()

    # flags
    f = g["secured_parties"].fillna("")
    flags = f.apply(secparty_flags).apply(pd.Series)
    g = pd.concat([g, flags], axis=1)

    # score
    W = cfg["scoring"]
    g["mca_score"] = (
        W["w_bank"] * g["has_bank"]
      + W["w_equipment"] * g["has_equipment"]
      + W["w_secparties"] * (g["count"].clip(0,10)/10.0)
      + W["w_filings"] * (g["active_filing_count"].clip(0,10)/10.0)
      - W["p_irs"] * g["has_irs"]
      - W["p_sba"] * g["has_sba"]
    )

    # filter negatives
    g = g[(g["has_irs"]==0) & (g["has_sba"]==0)]
    # drop empties
    g = g[(g["debtor_name"].str.len()>1) & (g["city"].astype(str).str.len()>0) & (g["zip"].astype(str).str.len()>0)]

    # dedupe across fuzzy variants of name+zip
    g["key"] = g["debtor_name"].str.lower().str.replace(r"[^a-z0-9 ]","", regex=True) + "|" + g["zip"].astype(str).str.upper()
    keep = []
    seen = set()
    g = g.sort_values(["mca_score","active_filing_count","count"], ascending=[False,False,False])
    for _, row in g.iterrows():
        if row["key"] in seen: continue
        keep.append(row)
        seen.add(row["key"])
    return pd.DataFrame(keep)

# ---------- Google Places enrichment ----------
PLACES_KEY = cfg["enrichment"]["google_places_api_key"]

def google_text_search(query):
    url = "https://maps.googleapis.com/maps/api/place/textsearch/json"
    r = requests.get(url, params={"query": query, "region": "us", "key": PLACES_KEY}, timeout=30)
    r.raise_for_status()
    return r.json()

def google_place_details(place_id):
    url = "https://maps.googleapis.com/maps/api/place/details/json"
    fields = "name,formatted_address,formatted_phone_number,international_phone_number,website,types,geometry,place_id"
    r = requests.get(url, params={"place_id": place_id, "fields": fields, "key": PLACES_KEY}, timeout=30)
    r.raise_for_status()
    return r.json()

def enrich_row(name, city):
    q = f"{name} {city} Colorado"
    try:
        s = google_text_search(q)
        cand = next((x for x in s.get("results",[]) if "colorado" in x.get("formatted_address","").lower()), None) or (s.get("results") or [None])[0]
        if not cand: return {}
        d = google_place_details(cand["place_id"])
        r = d.get("result",{})
        return {
            "place_id": cand["place_id"],
            "website": r.get("website",""),
            "phone": r.get("formatted_phone_number", r.get("international_phone_number","")),
            "address": r.get("formatted_address",""),
            "types": ",".join(r.get("types",[]))
        }
    except Exception:
        return {}

# ---------- OpenCorporates (optional) ----------
USE_OC = cfg["enrichment"].get("use_opencorporates", False)
OC_TOKEN = cfg["enrichment"].get("opencorporates_api_token","")

def opencorp_status(name):
    if not USE_OC or not OC_TOKEN: return {}
    url = "https://api.opencorporates.com/v0.4/companies/search"
    try:
        r = requests.get(url, params={"q": name, "api_token": OC_TOKEN, "jurisdiction_code": "us_co"}, timeout=30)
        r.raise_for_status()
        data = r.json()
        items = (data.get("results",{}) or {}).get("companies",[]) or []
        if not items: return {}
        c = items[0]["company"]
        return {
            "oc_company_number": c.get("company_number",""),
            "oc_status": c.get("current_status",""),
            "oc_jurisdiction": c.get("jurisdiction_code","")
        }
    except Exception:
        return {}

# ---------- main ----------
def main(top_n=200):
    # incremental watermark (you can implement date filters here if your dataset exposes them)
    debt = pull_debtors()
    sec = pull_secured()

    rolled = build_business_rollup(debt, sec)
    # skip already-seen businesses
    rolled = rolled[~rolled["key"].apply(is_seen)]
    rolled = rolled.sort_values("mca_score", ascending=False).head(top_n).copy()

    # enrich
    rows = []
    for _, r in rolled.iterrows():
        e = enrich_row(r["debtor_name"], r["city"])
        oc = opencorp_status(r["debtor_name"])
        rows.append({
            "debtor_name": r["debtor_name"],
            "address": e.get("address",""),
            "city": r["city"],
            "state": r["state"],
            "zip": r["zip"],
            "website": e.get("website",""),
            "phone": e.get("phone",""),
            "place_id": e.get("place_id",""),
            "sector": "heavy civil/construction" if r["has_equipment"] else "general contractor",
            "why_good_for_MCA": f"equipment/payroll intensive; banks={r['has_bank']}, equip={r['has_equipment']}, filings={r['active_filing_count']}, sp_count={r['count']}",
            "has_bank": r["has_bank"],
            "has_equipment": r["has_equipment"],
            "secured_party_count": r["count"],
            "active_filing_count": r["active_filing_count"],
            "last_filing_date": "" if pd.isna(r["last_filing_date"]) else str(r["last_filing_date"]),
            **oc
        })
        mark_seen(r["key"])

    out = pd.DataFrame(rows)
    out.to_csv(CSV_OUT, index=False)
    print(f"Wrote {len(out)} leads to {CSV_OUT}")

if __name__ == "__main__":
    top = int(os.environ.get("TOP_N","200"))
    main(top_n=top)
